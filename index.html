  <script>

    const nameInput = document.getElementById("name");

    const emailInput = document.getElementById("email");

    const phoneInput = document.getElementById("phone");

    const linkedinInput = document.getElementById("linkedin");

    const summaryInput = document.getElementById("summary");

    const experienceInput = document.getElementById("experience");

    const educationInput = document.getElementById("education");


    const skillInput = document.getElementById("skillInput");

    const addSkillBtn = document.getElementById("addSkillBtn");

    const skillsList = document.getElementById("skillsList");


    const rName = document.getElementById("rName");

    const rContact = document.getElementById("rContact");

    const rSummary = document.getElementById("rSummary");

    const rSkills = document.getElementById("rSkills");

    const rExperience = document.getElementById("rExperience");

    const rEducation = document.getElementById("rEducation");


    const aiImproveBtn = document.getElementById("aiImproveBtn");

    const clearBtn = document.getElementById("clearBtn");

    const aiBox = document.getElementById("aiBox");

    const aiText = document.getElementById("aiText");


    const templateSelect = document.getElementById("templateSelect");

    const resumePreview = document.getElementById("resumePreview");


    const colorThemeSelect = document.getElementById("colorThemeSelect");


    const photoUpload = document.getElementById("photoUpload");

    const photoPreview = document.getElementById("photoPreview");

    const resumePhoto = document.getElementById("resumePhoto");


    let skills = [];


    function updateResume() {

      const name = nameInput.value.trim() || "Your Name";

      const email = emailInput.value.trim() || "Email";

      const phone = phoneInput.value.trim() || "Phone";

      const linkedin = linkedinInput.value.trim() || "LinkedIn/Portfolio";


      rName.innerText = name;

      rContact.innerText = `${email} | ${phone} | ${linkedin}`;


      rSummary.innerText = summaryInput.value.trim() || "Your summary will appear here...";

      rExperience.innerText = experienceInput.value.trim() || "Your work experience will appear here...";

      rEducation.innerText = educationInput.value.trim() || "Your education details will appear here...";


      updateSkillsPreview();

    }


    function updateSkillsPreview() {

      rSkills.innerHTML = "";


      if (skills.length === 0) {

        rSkills.innerHTML = "<li>Add your skills...</li>";

        return;

      }


      skills.forEach(skill => {

        const li = document.createElement("li");

        li.innerText = skill;

        rSkills.appendChild(li);

      });

    }


    function addSkill() {

      const skill = skillInput.value.trim();

      if (skill === "") return;


      if (skills.includes(skill)) {

        skillInput.value = "";

        return;

      }


      skills.push(skill);

      skillInput.value = "";


      renderSkillTags();

      updateResume();

    }


    function renderSkillTags() {

      skillsList.innerHTML = "";


      skills.forEach((skill, index) => {

        const tag = document.createElement("div");

        tag.classList.add("skill-tag");


        tag.innerHTML = `

          ${skill}

          <button onclick="removeSkill(${index})">✕</button>

        `;


        skillsList.appendChild(tag);

      });

    }


    function removeSkill(index) {

      skills.splice(index, 1);

      renderSkillTags();

      updateResume();

    }


    function aiImprove() {

      let summary = summaryInput.value.trim();

      let exp = experienceInput.value.trim();


      let suggestions = "";


      if (summary.length < 50) {

        suggestions += "⚡ Add 2-3 lines in summary describing your experience, skills, and career goal.\n\n";

      }


      if (skills.length < 5) {

        suggestions += "⚡ Add at least 5 skills to make your resume ATS-friendly.\n\n";

      }


      if (exp.length < 60) {

        suggestions += "⚡ Work Experience should include achievements with numbers like: Increased sales by 30%.\n\n";

      }


      if (suggestions === "") {

        suggestions = "✅ Your resume looks professional. Add certifications and achievements to make it even stronger.";

      }


      if (summary === "") {

        summaryInput.value =

          "Highly motivated and results-driven professional with strong communication and leadership skills. Experienced in managing responsibilities efficiently, achieving targets, and delivering measurable business growth. Seeking a challenging role to contribute expertise and expand career development.";

      }


      aiBox.style.display = "block";

      aiText.innerText = suggestions;


      updateResume();

    }


    function clearAll() {

      nameInput.value = "";

      emailInput.value = "";

      phoneInput.value = "";

      linkedinInput.value = "";

      summaryInput.value = "";

      experienceInput.value = "";

      educationInput.value = "";


      photoUpload.value = "";

      photoPreview.src = "";

      resumePhoto.src = "";


      skills = [];

      renderSkillTags();


      aiBox.style.display = "none";

      updateResume();

    }


    function downloadPDF() {

      const options = {

        margin: 0.3,

        filename: "ResumeAI_Resume.pdf",

        image: { type: "jpeg", quality: 0.98 },

        html2canvas: { scale: 2 },

        jsPDF: { unit: "in", format: "a4", orientation: "portrait" }

      };


      html2pdf().set(options).from(resumePreview).save();

    }


    function switchTemplate() {

      const selected = templateSelect.value;

      resumePreview.classList.remove("modern", "classic", "minimal");

      resumePreview.classList.add(selected);

    }


    function switchThemeColor() {

      const selectedColor = colorThemeSelect.value;

      document.documentElement.style.setProperty("--primary", selectedColor);

    }


    function uploadPhoto(event) {

      const file = event.target.files[0];

      if (!file) return;


      const reader = new FileReader();

      reader.onload = function(e) {

        photoPreview.src = e.target.result;

        resumePhoto.src = e.target.result;

      };

      reader.readAsDataURL(file);

    }


    addSkillBtn.addEventListener("click", addSkill);


    skillInput.addEventListener("keypress", (e) => {

      if (e.key === "Enter") {

        e.preventDefault();

        addSkill();

      }

    });


    aiImproveBtn.addEventListener("click", aiImprove);

    clearBtn.addEventListener("click", clearAll);


    templateSelect.addEventListener("change", switchTemplate);

    colorThemeSelect.addEventListener("change", switchThemeColor);


    photoUpload.addEventListener("change", uploadPhoto);


    [nameInput, emailInput, phoneInput, linkedinInput, summaryInput, experienceInput, educationInput]

      .forEach(input => input.addEventListener("input", updateResume));


    updateResume();

    switchTemplate();

    switchThemeColor();

  </script>
